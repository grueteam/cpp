Первая программа на чистом C
Подключаем библиотеку: **stdio.h**
``` cpp
/* Hello World на C */
#include <stdio.h> /* Подключаем библиотеку */
// #include <iostream> // Для C++

int main() {
  printf("C: Hello world!\n");
  // std::cout << "Test" << std::endl;  // Для C++
  return 0;
}
```
``` cpp
// Hello World на C++
#include <iostream>

using namespace std;

int main() {
  cout << "C++: Hello world!" << endl;
  return 0;
}
```
